'use strict';

var _path    = require( 'path' );
var _fs      = require( 'fs' );

var template = require( 'peako/template' );

var regexps = {
  safe: '<%=\\s*([^]*?)\\s*%>',
  html: '<%-\\s*([^]*?)\\s*%>',
  code: '<%\\s*([^]*?)\\s*%>',
  comm: '<%#([^]*?)%>'
};

/**
 * @constructor T.ViewEngine
 * @param {object}  [options]
 * @param {object}  [options.settings]
 * @param {string}  [options.settings.layout]
 * @param {string}  [options.settings.views]
 * @param {boolean} [options.settings.cache]
 * @example
 * // Import.
 * var T = require( 'tmpl' );
 * // Create.
 * var engine = new T.ViewEngine();
 * // Create with initial settings.
 * var engine = new T.ViewEngine( {
 *   settings: {
 *     cache: process.env.NODE_ENV === 'production'
 *   }
 * } );
 * // Compatible with Express.
 * app.set( 'view engine', 'tmpl' );
 * // Exports "__express".
 * app.engine( 'tmpl', T.__express );
 */
function ViewEngine ( options )
{
  this.settings = {
    layout: 'layout',
    views: 'views',
    cache: false
  };

  if ( typeof options !== 'undefined' ) {
    Object.assign( this.settings, options.settings );
  }

  this._cache = {};
}

ViewEngine.prototype = {
  /**
   * @method T.ViewEngine#include
   * @param  {string} path
   * @param  {object} [data]
   * @return {string}
   * @example
   * <% data.posts.forEach( function ( post ) { %>
   *   <%- this.include( 'partials/post', { post: post } ) %>
   * <% }, this ); %>
   */
  include: function include ( path, data )
  {
    var source;

    if ( ! _path.extname( path ) ) {
      // default template source extension
      path += '.tmpl';
    }

    path = _path.resolve( this.settings.views, path );

    if ( ! this._cache[ path ] ) {
      this._cache[ path ] = template( _fs.readFileSync( path, 'UTF-8' ), { regexps: regexps } );
    } else if ( ! this.settings.cache ) {
      source = _fs.readFileSync( path, 'UTF-8' );

      // re-compile only when source changed
      if ( this._cache[ path ].source !== source ) {
        this._cache[ path ] = template( source, { regexps: regexps } );
      }
    }

    return this._cache[ path ].render.call( this, data || {} );
  },

  /**
   * @method T.ViewEngine#render
   * @param  {string} path
   * @param  {object} [data]
   * @return {string}
   * @example
   * let feed = engine.render( 'feed', { posts } );
   */
  render: function render ( path, data )
  {
    if ( ! data ) {
      data = {};
    }

    return this.include( this.settings.layout, {
      content: this.include( path, data ),
      data: data
    } );
  },

  /**
   * @method T.ViewEngine#set
   * @param  {string} setting
   * @param  {any}    value
   * @return {void}           Returns nothing.
   * @example
   * if ( process.env.NODE_ENV === 'production' ) {
   *   engine.set( 'cache', true );
   * }
   */
  set: function set ( setting, value )
  {
    if ( isUnknown( setting ) ) {
      throw Error( 'Got unknown setting key: ' + setting );
    }

    this.settings[ setting ] = value;
  },

  /**
   * @method T.ViewEngine#get
   * @param  {string} setting
   * @return {any}            A value of the setting.
   * @example
   * var views = engine.get( 'views' );
   */
  get: function get ( setting )
  {
    if ( isUnknown( setting ) ) {
      throw Error( 'Got unknown setting key: ' + setting );
    }

    return this.settings[ setting ];
  },

  constructor: ViewEngine
};

function isUnknown ( setting )
{
  return setting !== 'layout' && setting !== 'views' && setting !== 'cache';
}

module.exports = ViewEngine;
